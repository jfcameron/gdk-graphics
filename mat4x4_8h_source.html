88  template<typename high_precision_buffer_type = long double>
89  void rotate(const Quaternion<component_type> &aRotation)
90  {
91  static_assert(std::is_floating_point<high_precision_buffer_type>::value, "high_precision_buffer_type must be a floating point type");
92 
93  const Quaternion<component_type> &q = aRotation;
94 
95  high_precision_buffer_type sqw = q.w * q.w;
96  high_precision_buffer_type sqx = q.x * q.x;
97  high_precision_buffer_type sqy = q.y * q.y;
98  high_precision_buffer_type sqz = q.z * q.z;
99 
100  // invs (inverse square length) is only required if quaternion is not already normalised
101  high_precision_buffer_type invs = 1 / (sqx + sqy + sqz + sqw);
102 
103  m[0][0] = ( sqx - sqy - sqz + sqw) * invs ; // since sqw + sqx + sqy + sqz =1/invs*invs
104  m[1][1] = (-sqx + sqy - sqz + sqw) * invs ;
105  m[2][2] = (-sqx - sqy + sqz + sqw) * invs ;
106 
107  high_precision_buffer_type tmp1 = q.x * q.y;
108  high_precision_buffer_type tmp2 = q.z * q.w;
109 
110  m[1][0] = 2.0 * static_cast<component_type>(tmp1 + tmp2) * invs;
111  m[0][1] = 2.0 * static_cast<component_type>(tmp1 - tmp2) * invs;
112 
113  tmp1 = q.x * q.z;
114  tmp2 = q.y * q.w;
115 
116  m[2][0] = 2.0 * static_cast<component_type>(tmp1 - tmp2) * invs;
117  m[0][2] = 2.0 * static_cast<component_type>(tmp1 + tmp2) * invs;
118 
119  tmp1 = q.y * q.z;
120  tmp2 = q.x * q.w;
121 
122  m[2][1] = 2.0 * static_cast<component_type>(tmp1 + tmp2) * invs;
123  m[1][2] = 2.0 * static_cast<component_type>(tmp1 - tmp2) * invs;
124  }
