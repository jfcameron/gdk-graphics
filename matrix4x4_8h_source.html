48 void set_to_orthographic(
49 const vector2_type &aOrthoSize,
50 const component_type aNearClippingPlane,
51 const component_type aFarClippingPlane,
52 const component_type aViewportAspectRatio) {
53 const auto x = 2.0f / (aOrthoSize.x * aViewportAspectRatio);
54 const auto y = 2.0f / aOrthoSize.y;
55 const auto n = - (aFarClippingPlane + aNearClippingPlane) / (aFarClippingPlane - aNearClippingPlane);
56 const auto f = -2.0f / (aFarClippingPlane - aNearClippingPlane);
57
58 m[0][0] = x ; m[1][0] = 0.; m[2][0] = 0.; m[3][0] = 0.;
59 m[0][1] = 0.; m[1][1] = y ; m[2][1] = 0.; m[3][1] = 0.;
60 m[0][2] = 0.; m[1][2] = 0.; m[2][2] = f ; m[3][2] = n ;
61 m[0][3] = 0.; m[1][3] = 0.; m[2][3] = 0.; m[3][3] = 1.;
62 }

65 void set_to_perspective(
66 const component_type aFieldOfView,
67 const component_type aNearClippingPlane,
68 const component_type aFarClippingPlane,
69 const component_type aViewportAspectRatio) {
70 const auto tanHalfFovy = static_cast<component_type>(tan(aFieldOfView * 0.5));
71
72 m[0][0] = 1.0 / (aViewportAspectRatio * tanHalfFovy);
73 m[0][1] = 0.0;
74 m[0][2] = 0.0;
75 m[0][3] = 0.0;
76
77 m[1][0] = 0.0;
78 m[1][1] = 1.0 / tanHalfFovy;
79 m[1][2] = 0.0;
80 m[1][3] = 0.0;
81
82 m[2][0] = 0.0;
83 m[2][1] = 0.0;
84 m[2][2] =-(aFarClippingPlane + aNearClippingPlane) / (aFarClippingPlane - aNearClippingPlane);
85 m[2][3] =-1.0;
86
87 m[3][0] = 0.0;
88 m[3][1] = 0.0;
89 m[3][2] =-2.0 * aFarClippingPlane * aNearClippingPlane / (aFarClippingPlane - aNearClippingPlane);
90 m[3][3] = 0.0;

110 quaternion_type rotation() const {
111 vector3_type scale = this->scale();
112 std::array<component_type *, 3> scaleAsArray { &scale.x, &scale.y, &scale.z };
113
114 matrix4x4<component_type> normalizedMatrix = *this;
115 for (order_type i{0}; i < 3; ++i) {
116 for (order_type j{0}; j < 3; ++j) {
117 normalizedMatrix.m[i][j] /= *scaleAsArray[i];
118 }
119 }
120
121 component_type trace = normalizedMatrix.m[0][0] + normalizedMatrix.m[1][1] + normalizedMatrix.m[2][2];
122 quaternion_type q;
123
124 if (trace > 0) {
125 component_type s = 0.5 / std::sqrt(trace + 1.0);
126 q.w = 0.25 / s;
127 q.x = (normalizedMatrix.m[2][1] - normalizedMatrix.m[1][2]) * s;
128 q.y = (normalizedMatrix.m[0][2] - normalizedMatrix.m[2][0]) * s;
129 q.z = (normalizedMatrix.m[1][0] - normalizedMatrix.m[0][1]) * s;
130 }
131 else {
132 if (normalizedMatrix.m[0][0] > normalizedMatrix.m[1][1] && normalizedMatrix.m[0][0] > normalizedMatrix.m[2][2]) {
133 component_type s = 2.0 * std::sqrt(1.0 + normalizedMatrix.m[0][0] - normalizedMatrix.m[1][1] - normalizedMatrix.m[2][2]);
134 q.w = (normalizedMatrix.m[2][1] - normalizedMatrix.m[1][2]) / s;
135 q.x = 0.25 * s;
136 q.y = (normalizedMatrix.m[0][1] + normalizedMatrix.m[1][0]) / s;
137 q.z = (normalizedMatrix.m[0][2] + normalizedMatrix.m[2][0]) / s;
138 }
139 else if (normalizedMatrix.m[1][1] > normalizedMatrix.m[2][2]) {
140 component_type s = 2.0 * std::sqrt(1.0 + normalizedMatrix.m[1][1] - normalizedMatrix.m[0][0] - normalizedMatrix.m[2][2]);
141 q.w = (normalizedMatrix.m[0][2] - normalizedMatrix.m[2][0]) / s;
142 q.x = (normalizedMatrix.m[0][1] + normalizedMatrix.m[1][0]) / s;
143 q.y = 0.25 * s;
144 q.z = (normalizedMatrix.m[1][2] + normalizedMatrix.m[2][1]) / s;
145 }
146 else {
147 component_type s = 2.0 * std::sqrt(1.0 + normalizedMatrix.m[2][2] - normalizedMatrix.m[0][0] - normalizedMatrix.m[1][1]);
148 q.w = (normalizedMatrix.m[1][0] - normalizedMatrix.m[0][1]) / s;
149 q.x = (normalizedMatrix.m[0][2] + normalizedMatrix.m[2][0]) / s;
150 q.y = (normalizedMatrix.m[1][2] + normalizedMatrix.m[2][1]) / s;
151 q.z = 0.25 * s;
152 }
153 }
154
155 q.z = -q.z; // opengl right-handed convention
156
157 return q;
158 }

161 void set_rotation(const quaternion_type &aRotation, const vector3_type &aScale = {1}) {
162 using namespace std;
163
164 const quaternion_type &q = aRotation;
165
166 const auto sqw = q.w * q.w;
167 const auto sqx = q.x * q.x;
168 const auto sqy = q.y * q.y;
169 const auto sqz = q.z * q.z;
170
171 const auto invs = 1 / (sqx + sqy + sqz + sqw);
172
173 m[0][0] = ( sqx - sqy - sqz + sqw) * invs ;
174 m[1][1] = (-sqx + sqy - sqz + sqw) * invs ;
175 m[2][2] = (-sqx - sqy + sqz + sqw) * invs ;
176
177 auto tmp1 = q.x * q.y;
178 auto tmp2 = q.z * q.w;
179
180 m[1][0] = 2.0 * static_cast<component_type>(tmp1 + tmp2) * invs;
181 m[0][1] = 2.0 * static_cast<component_type>(tmp1 - tmp2) * invs;
182
183 tmp1 = q.x * q.z;
184 tmp2 = q.y * q.w;
185
186 m[2][0] = 2.0 * static_cast<component_type>(tmp1 - tmp2) * invs;
187 m[0][2] = 2.0 * static_cast<component_type>(tmp1 + tmp2) * invs;
188
189 tmp1 = q.y * q.z;
190 tmp2 = q.x * q.w;
191
192 m[2][1] = 2.0 * static_cast<component_type>(tmp1 + tmp2) * invs;
193 m[1][2] = 2.0 * static_cast<component_type>(tmp1 - tmp2) * invs;
194
195 m[0][0] *= aScale.x;
196 m[0][1] *= aScale.x;
197 m[0][2] *= aScale.x;
198
199 m[1][0] *= aScale.y;
200 m[1][1] *= aScale.y;
201 m[1][2] *= aScale.y;
202
203 m[2][0] *= aScale.z;
204 m[2][1] *= aScale.z;
205 m[2][2] *= aScale.z;
206 }

232 void inverse() {
233 component_type s0 = m[0][0] * m[1][1] - m[1][0] * m[0][1];
234 component_type s1 = m[0][0] * m[1][2] - m[1][0] * m[0][2];
235 component_type s2 = m[0][0] * m[1][3] - m[1][0] * m[0][3];
236 component_type s3 = m[0][1] * m[1][2] - m[1][1] * m[0][2];
237 component_type s4 = m[0][1] * m[1][3] - m[1][1] * m[0][3];
238 component_type s5 = m[0][2] * m[1][3] - m[1][2] * m[0][3];
239
240 component_type c5 = m[2][2] * m[3][3] - m[3][2] * m[2][3];
241 component_type c4 = m[2][1] * m[3][3] - m[3][1] * m[2][3];
242 component_type c3 = m[2][1] * m[3][2] - m[3][1] * m[2][2];
243 component_type c2 = m[2][0] * m[3][3] - m[3][0] * m[2][3];
244 component_type c1 = m[2][0] * m[3][2] - m[3][0] * m[2][2];
245 component_type c0 = m[2][0] * m[3][1] - m[3][0] * m[2][1];
246
247 component_type invdet = component_type(1) / (s0 * c5 - s1 * c4 + s2 * c3 + s3 * c2 - s4 * c1 + s5 * c0);
248
249 component_type b[order][order];
250
251 b[0][0] = ( m[1][1] * c5 - m[1][2] * c4 + m[1][3] * c3) * invdet;
252 b[0][1] = (-m[0][1] * c5 + m[0][2] * c4 - m[0][3] * c3) * invdet;
253 b[0][2] = ( m[3][1] * s5 - m[3][2] * s4 + m[3][3] * s3) * invdet;
254 b[0][3] = (-m[2][1] * s5 + m[2][2] * s4 - m[2][3] * s3) * invdet;
255
256 b[1][0] = (-m[1][0] * c5 + m[1][2] * c2 - m[1][3] * c1) * invdet;
257 b[1][1] = ( m[0][0] * c5 - m[0][2] * c2 + m[0][3] * c1) * invdet;
258 b[1][2] = (-m[3][0] * s5 + m[3][2] * s2 - m[3][3] * s1) * invdet;
259 b[1][3] = ( m[2][0] * s5 - m[2][2] * s2 + m[2][3] * s1) * invdet;
260
261 b[2][0] = ( m[1][0] * c4 - m[1][1] * c2 + m[1][3] * c0) * invdet;
262 b[2][1] = (-m[0][0] * c4 + m[0][1] * c2 - m[0][3] * c0) * invdet;
263 b[2][2] = ( m[3][0] * s4 - m[3][1] * s2 + m[3][3] * s0) * invdet;
264 b[2][3] = (-m[2][0] * s4 + m[2][1] * s2 - m[2][3] * s0) * invdet;
265
266 b[3][0] = (-m[1][0] * c3 + m[1][1] * c1 - m[1][2] * c0) * invdet;
267 b[3][1] = ( m[0][0] * c3 - m[0][1] * c1 + m[0][2] * c0) * invdet;
268 b[3][2] = (-m[3][0] * s3 + m[3][1] * s1 - m[3][2] * s0) * invdet;
269 b[3][3] = ( m[2][0] * s3 - m[2][1] * s1 + m[2][2] * s0) * invdet;
270
271 set(
272 b[0][0], b[0][1], b[0][2], b[0][3],
273 b[1][0], b[1][1], b[1][2], b[1][3],
274 b[2][0], b[2][1], b[2][2], b[2][3],
275 b[3][0], b[3][1], b[3][2], b[3][3]
276 );
277 }

281 void inverse_affine() {
282 component_type rot[3][3] = {
283 { m[0][0], m[0][1], m[0][2] },
284 { m[1][0], m[1][1], m[1][2] },
285 { m[2][0], m[2][1], m[2][2] }
286 };
287
288 component_type trans[3] = { m[3][0], m[3][1], m[3][2] };
289
290 component_type rotInv[3][3] = {
291 { rot[0][0], rot[1][0], rot[2][0] },
292 { rot[0][1], rot[1][1], rot[2][1] },
293 { rot[0][2], rot[1][2], rot[2][2] }
294 };
295
296 component_type transInv[3] = {
297 -(rotInv[0][0] * trans[0] + rotInv[0][1] * trans[1] + rotInv[0][2] * trans[2]),
298 -(rotInv[1][0] * trans[0] + rotInv[1][1] * trans[1] + rotInv[1][2] * trans[2]),
299 -(rotInv[2][0] * trans[0] + rotInv[2][1] * trans[1] + rotInv[2][2] * trans[2])
300 };
301
302 set(
303 rotInv[0][0], rotInv[0][1], rotInv[0][2], 0.0,
304 rotInv[1][0], rotInv[1][1], rotInv[1][2], 0.0,
305 rotInv[2][0], rotInv[2][1], rotInv[2][2], 0.0,
306 transInv[0], transInv[1], transInv[2], 1.0
307 );
308 }

430
431 aMatrix.m[0][1] = -aMatrix.m[0][1];
432 aMatrix.m[0][2] = -aMatrix.m[0][2];
433 aMatrix.m[1][0] = -aMatrix.m[1][0];
434 aMatrix.m[1][2] = -aMatrix.m[1][2];

436 aMatrix.m[2][1] = -aMatrix.m[2][1];
437
438 const auto newX = aMatrix.m[0][0] * vec.x + aMatrix.m[1][0] * vec.y + aMatrix.m[2][0] * vec.z + aMatrix.m[3][0];
439 const auto newY = aMatrix.m[0][1] * vec.x + aMatrix.m[1][1] * vec.y + aMatrix.m[2][1] * vec.z + aMatrix.m[3][1];
440 const auto newZ = aMatrix.m[0][2] * vec.x + aMatrix.m[1][2] * vec.y + aMatrix.m[2][2] * vec.z + aMatrix.m[3][2];
441 const auto newW = aMatrix.m[0][3] * vec.x + aMatrix.m[1][3] * vec.y + aMatrix.m[2][3] * vec.z + aMatrix.m[3][3];
442
443 if (std::abs(newW - 1.0) > 1e-6) {
444 aVector.x = newX / newW;
445 aVector.y = newY / newW;
446 aVector.z = newZ / newW;
447 } else {
448 aVector.x = newX;
449 aVector.y = newY;
450 aVector.z = newZ;
451 }
