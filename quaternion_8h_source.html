22 quaternion<component_type> normalized() const {
23 component_type magnitude = sqrt(pow(x, 2) + pow(y, 2) + pow(z, 2) + pow(w, 2));
24 if (magnitude == 0.0) {
25 //throw std::runtime_error("Cannot normalize a zero-length quaternion.");
26 return {};
27 }
28 float invMagnitude = 1.0f / magnitude;
29 return { x * invMagnitude, y * invMagnitude, z * invMagnitude, w * invMagnitude };
30 }

32 void set_from_euler(const vector3<component_type> &aEulerAngles) {
33 static const component_type HALF(0.5);
34
35 const auto yaw(aEulerAngles.z);
36 const auto pitch(aEulerAngles.y);
37 const auto roll(aEulerAngles.x);
38
39 using namespace std;
40
41 const auto cy = cos(yaw * HALF);
42 const auto sy = sin(yaw * HALF);
43 const auto cp = cos(pitch * HALF);
44 const auto sp = sin(pitch * HALF);
45 const auto cr = cos(roll * HALF);
46 const auto sr = sin(roll * HALF);
47
48 w = cr * cp * cy + sr * sp * sy;
49 x = sr * cp * cy - cr * sp * sy;
50 y = cr * sp * cy + sr * cp * sy;
51 z = cr * cp * sy - sr * sp * cy;
52 }

54 vector3<component_type> to_euler() const {
55 using namespace std;
56
57 const component_type sinr_cosp = 2 * (w * x + y * z);
58 const component_type cosr_cosp = 1 - 2 * (x * x + y * y);
59 const component_type roll = atan2(sinr_cosp, cosr_cosp);
60
61 const component_type sinp = 2 * (w * y - z * x);
62 const component_type pitch = asin(sinp);
63
64 const component_type siny_cosp = 2 * (w * z + x * y);
65 const component_type cosy_cosp = 1 - 2 * (y * y + z * z);
66 const component_type yaw = atan2(siny_cosp, cosy_cosp);
67
68 return {roll, pitch, yaw};
69 }

71 // Full inverse (use for non-unit quaternions)
72 quaternion<component_type> inverse() const {
73 component_type normSquared = x*x + y*y + z*z + w*w;
74 if (normSquared == 0.0f) {
75 throw std::runtime_error("Cannot invert a zero quaternion.");
76 }
77 component_type invNorm = 1.0f / normSquared;
78 return { -x * invNorm, -y * invNorm, -z * invNorm, w * invNorm };
79 }

133 quaternion<component_type> operator*(const quaternion<component_type> a, const quaternion<component_type> b) {
134 return {
135 a.w * b.x + a.x * b.w + a.y * b.z - a.z * b.y,
136 a.w * b.y - a.x * b.z + a.y * b.w + a.z * b.x,
137 a.w * b.z + a.x * b.y - a.y * b.x + a.z * b.w,
